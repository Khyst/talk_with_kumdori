.env 는 아래 형식을 따라 직접 생성해줘야 함


```
# KAKAO Developers API Key
# KAKAO Map, Search 등 서비스를 위한 API_KEY 입력
KAKAO_DEV_API_KEY=<YOUR_KAKAO_DEV_API_KEY>

# OpenAI API Key
# GPT Model 활용을 위한 입력
OPENAI_API_KEY=<YOUR_OPENAI_API_KEY>
OPENAI_DEFAULT_MODEL=<DEFAULT_MODEL_VER_YOU_WANT_TO_USE>

# LangSmith API Key
# RAG 컴포넌트 요소 호출 히스토리 관리를 위한 Langsmith 활성화를 위한 API Keys
LANGCHAIN_TRACING_V2=<YOUR_LANGCHAIN_TRACING_V2>
LANGCHAIN_ENDPOINT=<YOUR_LANGCHAIN_ENDPOINT>
LANGCHAIN_API_KEY=<YOUR_LANGCHAIN_API_KEY>
LANGCHAIN_PROJECT=<YOUR_LANGCHAIN_PROJECT>

# Tavily (https://app.tavily.com/)
# 실시간 검색을 위한 API 키 생성
TAVILY_API_KEY=<YOUR_TAVILY_API_KEY>

# Google Maps Platform (https://console.cloud.google.com/google/maps-apis/)
# - 구글 Places API 이용을 위한 API 키 생성
GOOGLE_MAPS_PLATFORM_API_KEY=<YOUR_GOOGLE_MAPS_PLATFORM_API_KEY>

# Weather API Key
# - K Weather Service 이용을 위한 API 키 생성
WEATHER_API_KEY=<YOUR_WEATHER_API_KEY>

# TMDB API Key
# - 실시간 영화 순위 및 정보 제공 서비스 (TMDB) API 이용을 위한 키 생성
TMDB_API_KEY=<YOUR_TMDB_API_KEY>

```
